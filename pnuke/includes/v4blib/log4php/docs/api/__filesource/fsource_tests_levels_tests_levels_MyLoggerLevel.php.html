62        /**
63         * Convert the string passed as argument to a level. If the
64         * conversion fails, then this method returns a TRACE Level.
65         *
66         * @param mixed $arg 
67         * @param LoggerLevel $defaultLevel 
68         * @static
69         */
70        function toLevel($arg, $defaultLevel = null)
71        {
72            if ($defaultLevel === null) {
73                return LoggerLevel::toLevel($arg, MyLoggerLevel::getLevelTrace());
74            } else {
75                if (is_int($arg)) {
76                    switch($arg) {
77                        case LOG4PHP_LEVEL_TRACE_INT:   return MyLoggerLevel::getLevelTrace();
78                        case LOG4PHP_LEVEL_LETHAL_INT:  return MyLoggerLevel::getLevelLethal();
79                        default:                        return LoggerLevel::toLevel($arg, $defaultLevel);
80                    }
81                } else {
82                    switch(strtoupper($arg)) {
83                        case 'TRACE':   return MyLoggerLevel::getLevelTrace();
84                        case 'LETHAL':  return MyLoggerLevel::getLevelLethal();
85                        default:        return LoggerLevel::toLevel($arg, $defaultLevel);
86                    }
87                }
88            }
